// ───────────────────────────────────────────────────────────────────── // VoiceRx Worker v2 — Whisper + Anthropic two-step pipeline // ───────────────────────────────────────────────────────────────────── // // SECRETS REQUIRED (Cloudflare dash → Worker → Settings → Variables): // ANTHROPIC_API_KEY ← sk-ant-… // CLIENT_TOKENS ← Dr. Tess's token (comma-separated if multiple) // OPENAI_API_KEY ← sk-… (for Whisper transcription) // // ENDPOINTS: // GET / → health check // POST /transcribe → multipart: audio file + token field // Returns { transcript, note } // POST / → legacy Anthropic passthrough (kept for compatibility) // // PHI handling: passthrough only. No storage, no logging. // Cloudflare Workers = no BAA needed (passthrough). // Anthropic BAA ✅ | OpenAI BAA: check before production PHI. // ───────────────────────────────────────────────────────────────────── const ALLOWED_ORIGIN = 'https://voicerx.pages.dev'; const ANTHROPIC_MODEL = 'claude-sonnet-4-5'; const MAX_TOKENS = 4096; const CORS = { 'Access-Control-Allow-Origin': ALLOWED_ORIGIN, 'Access-Control-Allow-Methods': 'POST, OPTIONS, GET', 'Access-Control-Allow-Headers': 'Content-Type', 'Access-Control-Max-Age': '86400', }; // ── Dr. Tess's system prompt ────────────────────────────────────────── const CLINICAL_SYSTEM_PROMPT = `You are a medical scribe assistant for Dr. Theresa Garcia, MD, a Direct Primary Care physician at Garcia Family Medicine in Blue Springs, MO. You receive raw transcripts from ambient patient encounter recordings. Your job is to clean and format them into structured clinical notes. FORMAT THE NOTE AS: **CHIEF COMPLAINT:** [What brought the patient in today] **HISTORY OF PRESENT ILLNESS:** [Narrative of the current problem — onset, duration, character, associated symptoms, what makes it better/worse] **REVIEW OF SYSTEMS:** [Pertinent positives and negatives mentioned] **ASSESSMENT:** [Dr. Garcia's clinical impressions and diagnoses mentioned] **PLAN:** [Treatments, prescriptions, referrals, follow-up, labs, imaging ordered] **PATIENT EDUCATION:** [Instructions given to the patient, if any] RULES: - Include ONLY what was actually said in the transcript. Never add clinical content that wasn't spoken. - If a section has no content from the transcript, write "Not documented this visit." - Clean up filler words (um, uh, like) and false starts naturally. - Preserve Dr. Garcia's exact clinical language and phrasing. - Flag anything that sounds incomplete or unclear with [REVIEW] tag. - This note requires physician review and approval before becoming a legal record.`; export default { async fetch(request, env) { // CORS preflight if (request.method === 'OPTIONS') { return new Response(null, { status: 204, headers: CORS }); } // Health check if (request.method === 'GET') { return new Response(JSON.stringify({ status: 'VoiceRx API v2: ok', whisper: true }), { status: 200, headers: { ...CORS, 'Content-Type': 'application/json' }, }); } if (request.method !== 'POST') { return json({ error: 'Method not allowed' }, 405); } const url = new URL(request.url); // ── Route: /transcribe (Whisper + Anthropic pipeline) ───────────── if (url.pathname === '/transcribe') { return handleTranscribe(request, env); } // ── Route: / (legacy Anthropic passthrough) ─────────────────────── return handleLegacy(request, env); }, }; // ───────────────────────────────────────────────────────────────────── // TRANSCRIBE — two-step: Whisper → Anthropic // ───────────────────────────────────────────────────────────────────── async function handleTranscribe(request, env) { // Validate secrets if (!env.OPENAI_API_KEY) return json({ error: 'Server not configured (missing OPENAI_API_KEY)' }, 500); if (!env.ANTHROPIC_API_KEY) return json({ error: 'Server not configured (missing ANTHROPIC_API_KEY)' }, 500); // Parse multipart form let formData; try { formData = await request.formData(); } catch { return json({ error: 'Expected multipart/form-data with audio file and token' }, 400); } // Token gate const token = formData.get('token'); const validTokens = String(env.CLIENT_TOKENS || '') .split(',') .map((t) => t.trim()) .filter(Boolean); if (!token || !validTokens.includes(token)) { return json({ error: 'Invalid or missing token' }, 401); } // Audio file const audioFile = formData.get('audio'); if (!audioFile || typeof audioFile === 'string') { return json({ error: 'audio field must be a file' }, 400); } // Optional metadata const patientName = formData.get('patientName') || 'Patient'; const noteType = formData.get('noteType') || 'SOAP'; const visitDate = formData.get('visitDate') || new Date().toLocaleDateString('en-US'); // ── Step 1: Whisper transcription ───────────────────────────────── const whisperForm = new FormData(); whisperForm.append('file', audioFile, 'encounter.webm'); whisperForm.append('model', 'whisper-1'); whisperForm.append('language', 'en'); whisperForm.append('response_format', 'text'); let transcript; try { const whisperRes = await fetch('https://api.openai.com/v1/audio/transcriptions', { method: 'POST', headers: { Authorization: `Bearer ${env.OPENAI_API_KEY}` }, body: whisperForm, }); if (!whisperRes.ok) { const errText = await whisperRes.text(); return json({ error: 'Whisper transcription failed', detail: errText }, 502); } transcript = await whisperRes.text(); if (!transcript || transcript.trim().length === 0) { return json({ error: 'No speech detected in recording' }, 422); } } catch (err) { return json({ error: 'Whisper connection failed', detail: String(err) }, 502); } // ── Step 2: Anthropic note formatting ───────────────────────────── const userMessage = `Patient: ${patientName} Visit Date: ${visitDate} Note Type: ${noteType} RAW TRANSCRIPT: ${transcript} Please format this into a structured clinical note.`; let formattedNote; try { const anthropicRes = await fetch('https://api.anthropic.com/v1/messages', { method: 'POST', headers: { 'Content-Type': 'application/json', 'x-api-key': env.ANTHROPIC_API_KEY, 'anthropic-version': '2023-06-01', }, body: JSON.stringify({ model: ANTHROPIC_MODEL, max_tokens: MAX_TOKENS, system: CLINICAL_SYSTEM_PROMPT, messages: [{ role: 'user', content: userMessage }], }), }); if (!anthropicRes.ok) { const errText = await anthropicRes.text(); return json({ error: 'Anthropic formatting failed', detail: errText }, 502); } const anthropicData = await anthropicRes.json(); formattedNote = anthropicData.content?.[0]?.text || ''; if (!formattedNote) { return json({ error: 'Anthropic returned empty response' }, 502); } } catch (err) { return json({ error: 'Anthropic connection failed', detail: String(err) }, 502); } // ── Return both transcript and formatted note ────────────────────── return json({ transcript, note: formattedNote, patientName, visitDate }); } // ───────────────────────────────────────────────────────────────────── // LEGACY — original Anthropic passthrough (unchanged) // ───────────────────────────────────────────────────────────────────── async function handleLegacy(request, env) { let body; try { body = await request.json(); } catch { return json({ error: 'Invalid JSON body' }, 400); } const { token, systemPrompt, messages } = body || {}; const validTokens = String(env.CLIENT_TOKENS || '') .split(',') .map((t) => t.trim()) .filter(Boolean); if (!token || !validTokens.includes(token)) { return json({ error: 'Invalid or missing token' }, 401); } if (!Array.isArray(messages) || messages.length === 0) { return json({ error: 'messages array required' }, 400); } if (!env.ANTHROPIC_API_KEY) { return json({ error: 'Server not configured (missing ANTHROPIC_API_KEY)' }, 500); } let upstream; try { upstream = await fetch('https://api.anthropic.com/v1/messages', { method: 'POST', headers: { 'Content-Type': 'application/json', 'x-api-key': env.ANTHROPIC_API_KEY, 'anthropic-version': '2023-06-01', }, body: JSON.stringify({ model: ANTHROPIC_MODEL, max_tokens: MAX_TOKENS, system: systemPrompt || '', messages, }), }); } catch (err) { return json({ error: 'Upstream connection failed', detail: String(err) }, 502); } const text = await upstream.text(); return new Response(text, { status: upstream.status, headers: { ...CORS, 'Content-Type': 'application/json' }, }); } function json(obj, status = 200) { return new Response(JSON.stringify(obj), { status, headers: { ...CORS, 'Content-Type': 'application/json' }, }); }
